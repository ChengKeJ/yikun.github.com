### i++
```c
int i=1;
int j;
j = i++;
```

```asm
mov         eax,dword ptr [i]  
mov         dword ptr [j],eax  ;将i值赋值给j
mov         ecx,dword ptr [i]  
add         ecx,1                    
mov         dword ptr [i],ecx  ;i自增
```

### ++i
```c
int i=1;
int j;
j = ++i;
```

```asm
mov         eax,dword ptr [i]  
add         eax,1  
mov         dword ptr [i],eax ;i自增 
mov         ecx,dword ptr [i]  
mov         dword ptr [j],ecx  ;将自增后i值给j
```
